    Fields:
    - Name text input
    - Description textarea (optional)
    - Path text input with live validation feedback:

      - Valid: show resolved path in green (e.g. "✓ /home/user/projects/my-app")
      - Invalid: show error message in red (e.g. "Directory not found")
    - Submit button disabled if path is invalid or form is loading

    Interface:
    ```typescript
    interface ProjectFormProps {
      initialData?: Project
      onSubmit: (data: ProjectInput) => void
      isLoading?: boolean
    }
    ```

    - On success: navigate to `/projects`
    - On error: show error message below form
    - Page title: "New Project" or "Edit Project"
