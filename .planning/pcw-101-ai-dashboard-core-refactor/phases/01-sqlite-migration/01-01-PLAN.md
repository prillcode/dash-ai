  <name>Task 2: Rewrite db/client.ts for better-sqlite3</name>
  <files>packages/server/src/db/client.ts</files>
  <action>
    Replace the entire file with a better-sqlite3 + drizzle-orm/better-sqlite3 implementation:

    ```ts
    import Database from "better-sqlite3"
    import { drizzle } from "drizzle-orm/better-sqlite3"
    import * as schema from "./schema"
    import path from "path"
    import os from "os"

    const dbPath = process.env.SQLITE_DB_PATH
      ? path.resolve(process.env.SQLITE_DB_PATH.replace(/^~/, os.homedir()))
      : path.join(os.homedir(), ".ai-dashboard", "dashboard.db")

    const sqlite = new Database(dbPath)

    // Enable WAL mode for better concurrent read performance
    sqlite.pragma("journal_mode = WAL")

    export const db = drizzle(sqlite, { schema })
    ```

    Key points:
    - Resolve ~ in SQLITE_DB_PATH just like projectService will do for project paths
    - Default to ~/.ai-dashboard/dashboard.db (auto-created by better-sqlite3 if missing)
    - WAL mode is important — without it, concurrent reads during queue polling will block writes
    - Import is ESM default import (better-sqlite3 exports a class as default)
  </action>
  <verify>pnpm --filter server tsc --noEmit 2>&1 | head -20</verify>
  <done>No TypeScript errors in db/client.ts, imports resolve correctly</done>
</task>

    Replace the entire file:

    ```ts
    import { defineConfig } from "drizzle-kit"
    import { config } from "dotenv"
    import path from "path"
    import os from "os"

    config({ path: path.resolve(__dirname, "../../.env") })

    const dbPath = process.env.SQLITE_DB_PATH
      ? path.resolve(process.env.SQLITE_DB_PATH.replace(/^~/, os.homedir()))
      : path.join(os.homedir(), ".ai-dashboard", "dashboard.db")

    export default defineConfig({
      schema: "./src/db/schema.ts",
      out: "./src/db/migrations",
      dialect: "sqlite",
      dbCredentials: {
        url: dbPath,
      },
    })
    ```

    Key points:
    - dialect changes from "turso" to "sqlite"
    - dbCredentials.url is the local file path (no authToken)
    - Use same path resolution logic as client.ts for consistency
    - __dirname is valid here because drizzle.config.ts is NOT ESM (it's CJS by default for drizzle-kit)
  </action>
  <verify>pnpm --filter server tsc --noEmit 2>&1 | head -20</verify>
  <done>No TypeScript errors, dialect is "sqlite", no Turso references remain in drizzle.config.ts</done>
