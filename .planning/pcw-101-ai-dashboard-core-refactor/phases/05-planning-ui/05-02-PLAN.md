    **Create `components/tasks/PlanDocViewer.tsx`:**
    ```tsx
    import { marked } from "marked"
    import { usePlanDoc } from "../../api/tasks"
    import { Spinner } from "../ui"

    interface PlanDocViewerProps {
      taskId: string
      file: "BRIEF.md" | "ROADMAP.md"
      title: string
    }

    export function PlanDocViewer({ taskId, file, title }: PlanDocViewerProps) {
      const { data, isLoading, error } = usePlanDoc(taskId, file, true)

      if (isLoading) {
        return (
          <div className="flex items-center gap-2 text-sm text-gray-500">
            <Spinner size="sm" /> Loading {file}...
          </div>
        )
      }

      if (error || !data) {
        return (
          <p className="text-sm text-gray-400 italic">
            {file} not yet generated.
          </p>
        )
      }

      const html = marked.parse(data.content) as string

      return (
        <details open className="border rounded">
          <summary className="px-3 py-2 cursor-pointer font-medium text-sm bg-gray-50 hover:bg-gray-100 rounded">
            {title}
          </summary>
          <div
            className="prose prose-sm max-w-none p-4 text-sm"
            dangerouslySetInnerHTML={{ __html: html }}
          />
        </details>
      )
    }
    ```

    Note: `dangerouslySetInnerHTML` is acceptable here — content comes from local
    files in the user's own repo, not from external/user-submitted sources.

    Check if Tailwind's `prose` plugin is enabled:
    ```bash
    cat packages/client/tailwind.config.js 2>/dev/null || cat packages/client/tailwind.config.ts 2>/dev/null
    ```
    If `@tailwindcss/typography` is not installed, add it:
    ```bash
    pnpm --filter client add -D @tailwindcss/typography
    ```
    And add `require("@tailwindcss/typography")` to `plugins` in tailwind config.
    If already present, skip.
