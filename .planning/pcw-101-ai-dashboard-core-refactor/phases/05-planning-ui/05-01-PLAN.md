    ```typescript
    // Start planning (DRAFT → IN_PLANNING)
    export function useStartPlanning() {
      const queryClient = useQueryClient()
      return useMutation({
        mutationFn: (taskId: string) =>
          apiClient<Task>(`/api/tasks/${taskId}/start-planning`, { method: "POST" }),
        onSuccess: (task) => {
          queryClient.invalidateQueries({ queryKey: ["task", task.id] })
          queryClient.invalidateQueries({ queryKey: ["tasks"] })
        },
      })
    }

    // Iterate plan (PLANNED → IN_PLANNING with feedback)
    export function useIteratePlan() {
      const queryClient = useQueryClient()
      return useMutation({
        mutationFn: ({ taskId, feedback }: { taskId: string; feedback: string }) =>
          apiClient<Task>(`/api/tasks/${taskId}/iterate-plan`, {
            method: "POST",
            body: JSON.stringify({ feedback }),
          }),
        onSuccess: (task) => {
          queryClient.invalidateQueries({ queryKey: ["task", task.id] })
        },
      })
    }

    // Read plan doc (BRIEF.md / ROADMAP.md)
    export function usePlanDoc(taskId: string, file: "BRIEF.md" | "ROADMAP.md" | "ISSUES.md", enabled: boolean) {
      return useQuery({
        queryKey: ["plan-doc", taskId, file],
        queryFn: () =>
          apiClient<{ file: string; content: string }>(
            `/api/tasks/${taskId}/plan-doc?file=${encodeURIComponent(file)}`
          ),
        enabled,
        staleTime: 30_000,
        retry: false,   // don't retry 404s (file doesn't exist yet)
      })
    }

    // Mark ready to code (PLANNED → READY_TO_CODE)
    export function useMarkReadyToCode() {
      const queryClient = useQueryClient()
      return useMutation({
        mutationFn: (taskId: string) =>
          apiClient<Task>(`/api/tasks/${taskId}/status`, {
            method: "PATCH",
            body: JSON.stringify({ status: "READY_TO_CODE" }),
          }),
        onSuccess: (task) => {
          queryClient.invalidateQueries({ queryKey: ["task", task.id] })
          queryClient.invalidateQueries({ queryKey: ["tasks"] })
        },
      })
    }
    ```
