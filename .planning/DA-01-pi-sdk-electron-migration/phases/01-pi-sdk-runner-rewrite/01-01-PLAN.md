# PLAN 01-01 — Install Pi SDK & Remove OpenCode Dependencies

## Objective
Swap `@opencode-ai/sdk` for `@mariozechner/pi-coding-agent` in the server package and remove all OpenCode-specific code paths.

## Tasks

### 1. Install Pi SDK
```bash
cd packages/server
pnpm remove @opencode-ai/sdk
pnpm add @mariozechner/pi-coding-agent @mariozechner/pi-ai @sinclair/typebox
```

### 2. Delete OpenCode auth check
- **File:** `packages/server/src/agent/authCheck.ts`
- **Action:** Delete the entire file
- It reads `~/.local/share/agent/auth.json` — replaced by Pi's `AuthStorage`

### 3. Create new Pi adapter module
- **File:** `packages/server/src/agent/piSession.ts` (new)
- **Action:** Create a shared module with:
  - `createPiAuth()` — initializes `AuthStorage.create()`
  - `createPiModelRegistry(auth)` — initializes `ModelRegistry.create(auth)`
  - `resolveModel(modelRegistry, provider, modelId)` — looks up a model, throws with clear error if not found or no API key
  - `checkPiAuth(auth, provider)` — validates provider has credentials
  - Export singleton instances (or lazy init) so auth/registry are shared across sessions

```typescript
import { AuthStorage, ModelRegistry } from "@mariozechner/pi-coding-agent"

let authStorage: AuthStorage | undefined
let modelRegistry: ModelRegistry | undefined

export function getAuth(): AuthStorage {
  if (!authStorage) authStorage = AuthStorage.create()
  return authStorage
}

export function getModelRegistry(): ModelRegistry {
  if (!modelRegistry) modelRegistry = ModelRegistry.create(getAuth())
  return modelRegistry
}

export async function checkProviderAuth(providerID: string): Promise<{ ok: boolean; errorMessage?: string }> {
  const registry = getModelRegistry()
  const available = await registry.getAvailable()
  const found = available.some(m => m.provider === providerID)
  if (!found) {
    return { ok: false, errorMessage: `No API key for provider "${providerID}". Set env var or run pi and use /login.` }
  }
  return { ok: true }
}

export async function resolveModel(provider: string, modelId: string) {
  const registry = getModelRegistry()
  const model = registry.find(provider, modelId)
  if (!model) throw new Error(`Model not found: ${provider}/${modelId}`)
  return model
}
```
